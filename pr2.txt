check
{
	int prev;
	bool is_ok = True;
	int input[255];
	int c;
	print("how many items? :")
	read(c);
	for (integer i = 0; i < c ;i++)
	{
		print("insert number")
		read(input[i]);
	}
	prev = input[0];
	
	for (integer i = 1; i < c; i++)
	{
		if(input[i] >= prev)
		{
			prev = input[i]
		}
		else
		{
			is_ok = False;
			i = c;
		}
	}
	
	if (is_ok)
		print("Input is ascending")
	else
		print("Input is NOT ascending")
	
}
